Step 1: The Core Agent

The foundational building block in ADK is the Agent. Let's create a simple agent that responds to customer queries.

step1_basic_agent.py
import asyncio
from google.adk import Agent

# Define the Agent
support_agent = Agent(
    name="support_agent",
    model="gemini-2.5-flash",
    instruction="You are a helpful customer support agent. Be polite and concise."
)

async def main():
    # Execute the agent synchronously via the async interface
    response = await support_agent.run("Hi, I need help with my recent order.")
    print(f"Agent Response: {response}")

if __name__ == "__main__":
    asyncio.run(main())

Step 2: Tools and Isolated State Management

Production agents need to interact with external APIs (like databases). Furthermore, because multiple users might interact with the agent simultaneously, we must isolate state using ToolContext.

step2_stateful_tools.py
import asyncio
from google.adk import Agent
from google.adk.tools import ToolContext

# Define a tool with Context Injection
def check_order_status(order_id: str, ctx: ToolContext) -> str:
    """Fetches the status of a customer order."""
    
    # 1. State Isolation: Read from the current user's session state
    cached_status = ctx.get_state(f"order_{order_id}", None)
    if cached_status:
        return f"Cached status: {cached_status}"
    
    # 2. Simulate database lookup
    status = "Shipped" if order_id.startswith("ORD-") else "Unknown"
    
    # 3. State Isolation: Write back to the session state safely
    ctx.set_state(f"order_{order_id}", status)
    
    return f"Database status: {status}"

# Attach the tool to the agent
support_agent = Agent(
    name="support_agent",
    model="gemini-2.5-flash",
    instruction="Help customers with their orders using tools.",
    tools=[check_order_status]
)

Step 3: Multi-Agent Workflow & Human-in-the-Loop (HITL)

When a customer asks for a refund, we want a Workflow that routes the request to a refund_agent. Before processing the refund, the workflow must PAUSE and wait for a human manager to approve it using a RequestInput node.

step3_workflow_hitl.py
from google.adk import Agent, Workflow
from google.adk.workflows import RequestInput

# 1. Define the Agents
triage_agent = Agent(
    name="triage_agent",
    instruction="Route customer tickets. If they want a refund, extract amount."
)

refund_agent = Agent(
    name="refund_agent",
    instruction="Execute the refund process based on the approved amount."
)

# 2. Define the Human Approval Gate
approval_gate = RequestInput(
    name="manager_approval",
    prompt_message="A refund has been requested. Type 'APPROVE' or 'DENY':"
)

# 3. Construct the Directed Graph Workflow
refund_workflow = Workflow(
    name="refund_workflow",
    edges=[
        ("START", triage_agent),
        (triage_agent, approval_gate),
        (approval_gate, refund_agent),
        (refund_agent, "END")
    ]
)

Step 4: Production Persistence (Vertex AI)

To ensure the agent remembers conversations across server restarts and maintains long-term knowledge about customers, replace the default memory store with Vertex AI Session Service and Vertex AI Memory Bank.

step4_production_vertexai.py
from google.adk.memory import VertexAiMemoryBankService
from google.adk.sessions import VertexAiSessionService
from google.adk.runtime import AgentRuntime
from step1_basic_agent import support_agent
import os

# 1. Configuration
PROJECT_ID = os.getenv("GCP_PROJECT_ID", "your-gcp-project-id")
LOCATION = os.getenv("GCP_LOCATION", "us-central1")
AGENT_ENGINE_ID = os.getenv("VERTEX_AGENT_ENGINE_ID", "your-agent-engine-id")

# 2. Initialize Vertex AI Session Service (Short-term context persistence)
session_service = VertexAiSessionService(
    project_id=PROJECT_ID,
    location=LOCATION,
    agent_engine_id=AGENT_ENGINE_ID
)

# 3. Initialize Vertex AI Memory Bank (Long-term semantic memory)
memory_service = VertexAiMemoryBankService(
    project=PROJECT_ID,
    location=LOCATION,
    agent_engine_id=AGENT_ENGINE_ID
)

# 4. Attach Vertex services to the ADK Runtime
runtime = AgentRuntime(
    session_service=session_service,
    memory_service=memory_service,  # Adds long-term memory backend
    default_model="gemini-2.5-flash"
)

# Register your agent to the runtime
runtime.register(support_agent)

Step 5: CI/CD Tool Trajectory Evaluation

Finally, to guarantee production reliability, write tests ensuring the agent uses tools correctly before deploying.

step5_evaluation.py
from google.adk.evaluation import AgentEvaluator, GoldenTestCase
from step2_stateful_tools import support_agent

# 1. Define Golden Datasets
test_cases = [
    GoldenTestCase(
        input_query="Check order ORD-999 please.",
        expected_output="Shipped",
        expected_tool_trajectory=[
            # Asserts exact tool and arguments
            ("check_order_status", {"order_id": "ORD-999"})
        ]
    )
]

def run_tests():
    # 2. Run the evaluator
    evaluator = AgentEvaluator(agent=support_agent)
    report = evaluator.run(test_cases)
    print(f"Tests Passed: {report.accuracy_score * 100}%")
