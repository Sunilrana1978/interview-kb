FastAPI
REST API Async
pip install fastapi uvicorn

The de-facto standard for building high-performance async REST APIs in Python. Built on Starlette + Pydantic, it auto-generates OpenAPI/Swagger docs, handles request validation via type hints, and achieves near Node.js throughput using ASGI + asyncio. Ideal for microservice APIs, ML model serving, and data APIs.

ML model serving REST microservices OpenAPI auto-docs Background tasks WebSockets
main.py 
from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel
from typing import Annotated

app = FastAPI(title="Product API", version="1.0")

class Product(BaseModel):
    name: str
    price: float
    sku: str

# Dependency injection — e.g., DB session or auth

    return {"user_id": "u123"}

@app.post("/products", status_code=201)

    product: Product,

pip install httpx

A fully async-capable HTTP client with an API nearly identical to Requests. Supports HTTP/2, connection pooling, timeouts, retries, and both sync and async modes. The preferred choice for calling external APIs inside FastAPI services and for testing APIs via TestClient.

Calling external APIs Async HTTP API testing HTTP/2 support
client.py
import httpx
import asyncio

# Async client with connection pooling
async def fetch_products(ids: list[str]) -> list[dict]:
    async with httpx.AsyncClient(
        base_url="https://api.shop.com",
        timeout=10.0,
        headers={"Authorization": "Bearer TOKEN"}
    ) as client:
        tasks = [client.get(f"/products/{i}") for i in ids]
        responses = await asyncio.gather(*tasks)
        return [r.json() for r in responses if r.status_code == 200]

producer = Producer({"bootstrap.servers": "kafka:9092"})

def publish_order_event(order: dict):

        topic="order.created",
        key=order["order_id"].encode(),

# product.proto (generate stubs: python -m grpc_tools.protoc ...)
# service ProductService { rpc GetProduct(ProductRequest) returns (ProductResponse); }

import grpc
from concurrent import futures
import product_pb2, product_pb2_grpc

class ProductServicer(product_pb2_grpc.ProductServiceServicer):
    def GetProduct(self, request, context):
        # request.product_id is type-safe via protobuf
        return product_pb2.ProductResponse(
            product_id=request.product_id,
            name="Air Force 1",
            price=110.00
        )

def serve():
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    product_pb2_grpc.add_ProductServiceServicer_to_server(ProductServicer(), server)
    server.add_insecure_port("[::]:50051")
    server.start()
    server.wait_for_termination()

import redis.asyncio as redis
import json
from functools import wraps

r = redis.Redis(host="localhost", port=6379, decode_responses=True)

# Cache decorator for FastAPI endpoints
def cache(ttl: int = 300):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            key = f"cache:{func.__name__}:{args}:{kwargs}"
            cached = await r.get(key)
            if cached: return json.loads(cached)
            result = await func(*args, **kwargs)
            await r.setex(key, ttl, json.dumps(result))
            return result
        return wrapper
    return decorator

# Distributed lock
async def process_with_lock(resource_id: str):
    lock = r.lock(f"lock:{resource_id}", timeout=30)
    async with lock:
        # Only one instance runs this block
        pass

import uuid

engine = create_async_engine("postgresql+asyncpg://user:pass@db/shop")

class Base(DeclarativeBase): pass

class Product(Base):
    __tablename__ = "products"
    id:    Mapped[str]   = mapped_column(String, primary_key=True, default=lambda: str(uuid.uuid4()))

async def get_products_over(min_price: float, session: AsyncSession):
    stmt = select(Product).where(Product.price > min_price).limit(50)
    result = await session.execute(stmt)
    return result.scalars().all()

The foundation of Python data analysis. Pandas provides the DataFrame — a 2D labeled table — for loading, cleaning, transforming, aggregating, and exporting structured data. Still the most-used library for ETL pipelines, EDA, and data prep for ML. Pandas 2.0 added Apache Arrow backend for significant speed gains.

etl.py
import pandas as pd

# Load → clean → transform → export
df = pd.read_parquet("s3://bucket/orders/2024/*.parquet",
                     storage_options={"anon": False})

# Clean
df = df.dropna(subset=["order_id", "amount"])
df["order_date"] = pd.to_datetime(df["order_date"])
df["amount"] = df["amount"].clip(lower=0)

# Feature engineering
df["month"] = df["order_date"].dt.month
df["is_high_value"] = df["amount"] > 500

# Aggregation
monthly = (df.groupby(["region", "month"])
             .agg(total_revenue=("amount", "sum"),
                  order_count=("order_id", "count"))
             .reset_index())

monthly.to_parquet("s3://bucket/output/monthly_revenue.parquet")

Polars
DataFramesHigh Performance
pip install polars

The modern, Rust-powered replacement for Pandas in performance-critical workloads. Polars uses columnar memory (Arrow), lazy evaluation, and automatic query optimization to process data 5–50× faster than Pandas. Its lazy API lets you build entire query plans before execution — similar to Spark but single-node and zero-overhead.

Large-file ETL Lazy query plans Multi-threaded transforms Parquet / Arrow pipelines
pipeline.py
import polars as pl

# Lazy API — nothing runs until .collect()
result = (
    pl.scan_parquet("orders/*.parquet")  # lazy scan
      .filter(pl.col("amount") > 0)
      .with_columns([
          pl.col("order_date").str.to_date().alias("date"),
          (pl.col("amount") * 1.1).alias("amount_with_tax"),
      ])
      .group_by(["region", pl.col("date").dt.month().alias("month")])
      .agg([
          pl.sum("amount_with_tax").alias("revenue"),
          pl.count("order_id").alias("orders"),
      ])
      .sort("revenue", descending=True)
      .collect()  # executes entire optimized plan
)

print(result.head(10))

import numpy as np

# Cosine similarity — used in RAG retrieval
def cosine_similarity(a: np.ndarray, b: np.ndarray) -> float:
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

query_embedding   = np.random.randn(1536).astype("float32")
doc_embeddings    = np.random.randn(1000, 1536).astype("float32")

# Vectorized batch similarity (no Python loop)
norms = np.linalg.norm(doc_embeddings, axis=1, keepdims=True)
normed = doc_embeddings / norms
scores = normed @ query_embedding / np.linalg.norm(query_embedding)
top5_idx = np.argsort(scores)[::-1][:5]

print(f"Top-5 doc indices: {top5_idx}")
print(f"Top-5 scores: {scores[top5_idx].round(4)}")
