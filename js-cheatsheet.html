Variables
const name = "Alice";     // immutable binding
let   age  = 30;         // block-scoped, mutable
var   old  = "avoid";    // function-scoped ⚠️

Operators
Equality
1 ==  "1"   // true  (loose, coerces)
1 === "1"   // false (strict) ✅ use this
1 !== 2     // true
Logical
true  && false   // false  (AND)
true  || false   // true   (OR)
!true            // false  (NOT)
null  ?? "def"   // "def"  (nullish)
a     ??= "val"  // assign if null/undef
Optional chaining
user?.address?.city  // no error if null
arr?.[0]              // safe index access
fn?.()               // call only if fn exists
Strings
const s = "Hello, World!";

s.length              // 13
s.toUpperCase()       // "HELLO, WORLD!"
s.toLowerCase()       // "hello, world!"
s.includes("World")   // true
s.startsWith("He")  // true
s.endsWith("!")     // true
s.slice(0, 5)        // "Hello"
s.replace("World","JS") // "Hello, JS!"
s.split(", ")        // ["Hello", "World!"]
s.trim()              // removes whitespace
s.padStart(15, "*") // "**Hello, World!"
s.repeat(2)          // "Hello...Hello..."

Arrays
Mutation methods
const a = [1, 2, 3];
a.push(4)       // [1,2,3,4] add end
a.pop()         // [1,2,3]   rem end
a.unshift(0)   // [0,1,2,3] add start
a.shift()       // [1,2,3]   rem start
a.splice(1,1)  // remove at index 1
a.reverse()     // [3,2,1] in-place
a.sort((x,y)=>x-y) // numeric sort
Spread & combine
const b = [4,5];
[...a, ...b]    // [1,2,3,4,5]
a.concat(b)     // [1,2,3,4,5]
a.flat()        // flatten 1 level
a.flat(Infinity) // flatten all
Iteration / transform
[1,2,3].map(x => x * 2)
// [2, 4, 6]

[1,2,3,4].filter(x => x % 2 === 0)
// [2, 4]

[1,2,3].reduce((acc, x) => acc + x, 0)
// 6

[1,2,3].find(x => x > 1)   // 2
[1,2,3].findIndex(x=>x>1) // 1
[1,2,3].some(x => x > 2)   // true
[1,2,3].every(x => x > 0)  // true
[1,2,3].includes(2)        // true
[1,2,3].forEach(x=>console.log(x))
[1,2,3].flatMap(x=>[x,x*2])
// [1,2, 2,4, 3,6]
Objects
const obj = { name: "Alice", age: 30 };

// Access
obj.name          // "Alice"
obj["age"]        // 30

// Keys / values / entries
Object.keys(obj)    // ["name","age"]
Object.values(obj)  // ["Alice",30]
Object.entries(obj) // [["name","Alice"],...]

// Merge / clone
const merged = { ...obj, city: "NYC" };
Object.assign({}, obj); // shallow clone

// Check key exists
"name" in obj    // true
obj.hasOwnProperty("name") // true

// Freeze (immutable)
Object.freeze(obj);
Functions
// Declaration (hoisted)
function add(a, b) { return a + b; }

// Expression
const add = function(a, b) { return a + b; };

// Arrow function ✅ preferred
const add = (a, b) => a + b;
const sq  = x => x ** 2;   // single param
const hi  = () => "hello"; // no params

// Default params
const greet = (name = "World") => `Hi ${name}`;

// Rest params
const sum = (...nums) => nums.reduce((a,b)=>a+b, 0);
Classes
class Animal {
  // class field
  sound = "...";

  constructor(name) {
    this.name = name;
  }

  speak() {
    return `${this.name} says ${this.sound}`;
  }

  static create(name) { return new Animal(name); }
}

class Dog extends Animal {
  sound = "Woof";

  constructor(name, breed) {
    super(name);        // call parent
    this.breed = breed;
  }
}

const d = new Dog("Rex", "Lab");
d.speak(); // "Rex says Woof"
d instanceof Animal; // true
Control Flow
// if / else
if (x > 0) { /* ... */ }
else if (x === 0) { /* ... */ }
else { /* ... */ }

// Ternary
const label = x > 0 ? "pos" : "neg";

// switch
switch (day) {
  case "Mon": break;
  default:   break;
}

// Loops
for (let i = 0; i < 3; i++) {}
for (const val of [1,2,3]) {}  // values
for (const key in obj) {}       // keys
while (cond) {}
do {} while (cond);

// Error handling
try {
  throw new Error("oops");
} catch (e) {
  console.error(e.message);
} finally { /* always runs */ }
Async / Await & Promises
Promise basics
const p = new Promise((resolve, reject) => {
  if (ok) resolve("data");
  else   reject(new Error("fail"));
});

p.then(data => console.log(data))
 .catch(err => console.error(err))
 .finally(() => console.log("done"));

// Promise combinators
Promise.all([p1, p2])        // all resolve
Promise.allSettled([p1,p2]) // all finish
Promise.race([p1, p2])       // first wins
Promise.any([p1, p2])        // first success
async / await ✅ preferred
async function fetchUser(id) {
  try {
    const res  = await fetch(`/api/${id}`);
    const data = await res.json();
    return data;
  } catch (e) {
    console.error("Error:", e);
  }
}

// Parallel await
const [a, b] = await Promise.all([
  fetchUser(1),
  fetchUser(2)
]);

// Top-level await (ES2022, modules)
const data = await fetchUser(1);
Destructuring
Array destructuring
const [a, b, c] = [1, 2, 3];
const [x, , z] = [1, 2, 3];    // skip index 1
const [h, ...rest] = [1,2,3]; // rest=[2,3]
let p = 1, q = 2;
[p, q] = [q, p];               // swap!
Object destructuring
const { name, age } = user;
const { name: n } = user;        // rename
const { city = "NYC" } = user;  // default
const { a: { b } } = nested;    // nested

// In function params
function greet({ name, age = 0 }) {
  return `${name} is ${age}`;
}
