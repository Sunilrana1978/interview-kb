Architecture

How it fits together

The chatbot uses Retrieval-Augmented Generation (RAG) via AWS Bedrock Knowledge Base. When a support engineer asks a question, the agent retrieves relevant document chunks from the indexed knowledge base, then passes them as context to a foundation model (Claude 3 or Amazon Titan) to generate a grounded response. Every interaction in this guide is driven by boto3 — no AWS CLI required.

Initial Setup

1Create the IAM role for Bedrock

Why boto3 for IAM?

Creating the role in Python keeps your infrastructure reproducible and version-controlled alongside your application code. The role grants Bedrock permission to read your S3 bucket and access OpenSearch Serverless.

Python · boto3 — create_bedrock_role.py 
import boto3
import json

iam = boto3.client('iam')
ACCOUNT_ID = boto3.client('sts').get_caller_identity()['Account']
BUCKET_NAME = 'prod-support-kb-docs'

# 1. Trust policy — allows Bedrock service to assume this role
trust_policy = {
    "Version": "2012-10-17",
    "Statement": [{
        "Effect": "Allow",
        "Principal": {"Service": "bedrock.amazonaws.com"},
        "Action": "sts:AssumeRole",
        "Condition": {
            "StringEquals": {
                "aws:SourceAccount": ACCOUNT_ID
            }
        }
    }]
}

# 2. Create the role
role = iam.create_role(
    RoleName='BedrockAgentRole',
    AssumeRolePolicyDocument=json.dumps(trust_policy),
    Description='Execution role for Bedrock Agent and Knowledge Base'
)
role_arn = role['Role']['Arn']
print(f"Role ARN: {role_arn}")

# 3. Inline permission policy
permissions = {
    "Version": "2012-10-17",
    "Statement": [
        {
            "Effect": "Allow",
            "Action": [
                "bedrock:InvokeModel",
                "bedrock:InvokeAgent",
                "bedrock:Retrieve",
                "bedrock:StartIngestionJob",
                "bedrock:GetIngestionJob",
                "bedrock:ListIngestionJobs"
            ],
            "Resource": "*"
        },
        {
            "Effect": "Allow",
            "Action": ["s3:GetObject", "s3:ListBucket"],
            "Resource": [
                f"arn:aws:s3:::{BUCKET_NAME}",
                f"arn:aws:s3:::{BUCKET_NAME}/*"
            ]
        },
        {
            "Effect": "Allow",
            "Action": ["aoss:APIAccessAll"],
            "Resource": "*"
        }
    ]
}

iam.put_role_policy(
    RoleName='BedrockAgentRole',
    PolicyName='BedrockAgentPermissions',
    PolicyDocument=json.dumps(permissions)
)
print("Permissions attached.")

2Create the S3 bucket and upload documents

Supported document formats

PDF, plain text (.txt), Markdown (.md), HTML, and Word (.docx). For best chunking quality use structured Markdown. Keep incident reports in incident-history/, feature docs in app-features/, and runbooks in runbooks/.

Python · boto3 — setup_s3_bucket.py 
import boto3
import os
from pathlib import Path

REGION      = 'us-east-1'
BUCKET_NAME = 'prod-support-kb-docs'
LOCAL_DOCS  = './knowledge-base'  # your local docs folder

s3 = boto3.client('s3', region_name=REGION)

# 1. Create bucket
try:
    if REGION == 'us-east-1':
        s3.create_bucket(Bucket=BUCKET_NAME)
    else:
        s3.create_bucket(
            Bucket=BUCKET_NAME,
            CreateBucketConfiguration={'LocationConstraint': REGION}
        )
    print(f"Bucket '{BUCKET_NAME}' created.")
except s3.exceptions.BucketAlreadyOwnedByYou:
    print("Bucket already exists — continuing.")

# 2. Enable versioning (recommended)
s3.put_bucket_versioning(
    Bucket=BUCKET_NAME,
    VersioningConfiguration={'Status': 'Enabled'}
)

# 3. Enable EventBridge notifications (needed for Phase 2)
s3.put_bucket_notification_configuration(
    Bucket=BUCKET_NAME,
    NotificationConfiguration={'EventBridgeConfiguration': {}}
)

# 4. Upload all local documents preserving folder structure
SKIP_EXTENSIONS = {'.DS_Store', '.tmp', '.gitkeep'}

def upload_documents(local_dir: str, bucket: str):
    base = Path(local_dir)
    uploaded = 0
    for filepath in base.rglob('*'):
        if filepath.is_file() and filepath.suffix not in SKIP_EXTENSIONS:
            s3_key = str(filepath.relative_to(base))
            s3.upload_file(str(filepath), bucket, s3_key)
            print(f"  Uploaded: {s3_key}")
            uploaded += 1
    print(f"Upload complete — {uploaded} files.")

upload_documents(LOCAL_DOCS, BUCKET_NAME)

3Create the Bedrock Knowledge Base

This script provisions an OpenSearch Serverless collection (vector store), creates the Knowledge Base, registers the S3 data source, and runs the first ingestion job — all via boto3.

Python · boto3 — create_knowledge_base.py 
import boto3
import json
import time

REGION      = 'us-east-1'
BUCKET_NAME = 'prod-support-kb-docs'
ROLE_ARN    = 'arn:aws:iam::ACCOUNT_ID:role/BedrockAgentRole'

bedrock_agent = boto3.client('bedrock-agent', region_name=REGION)
aoss          = boto3.client('opensearchserverless', region_name=REGION)
account_id    = boto3.client('sts').get_caller_identity()['Account']

# ── Step A: Create OpenSearch Serverless collection ──────────────────────
encryption_policy = {
    "Rules": [{"Resource": ["collection/prod-support-kb"],
               "ResourceType": "collection"}],
    "AWSOwnedKey": True
}
aoss.create_security_policy(
    name='prod-support-kb-enc',
    type='encryption',
    policy=json.dumps(encryption_policy)
)

network_policy = [{
    "Rules": [{"Resource": ["collection/prod-support-kb",
                            "dashboards/default"],
               "ResourceType": "collection"}],
    "AllowFromPublic": True
}]
aoss.create_security_policy(
    name='prod-support-kb-net',
    type='network',
    policy=json.dumps(network_policy)
)

collection = aoss.create_collection(
    name='prod-support-kb',
    type='VECTORSEARCH',
    description='Vector store for production support KB'
)
collection_id  = collection['createCollectionDetail']['id']
collection_arn = collection['createCollectionDetail']['arn']
print(f"Collection creating: {collection_id}")

# Wait for collection to become ACTIVE
while True:
    status = aoss.batch_get_collection(ids=[collection_id])['collectionDetails'][0]['status']
    print(f"  Collection status: {status}")
    if status == 'ACTIVE':
        break
    time.sleep(15)

# Data access policy — allow Bedrock role to read/write the index
data_policy = [{
    "Rules": [
        {"Resource": ["collection/prod-support-kb"],
         "Permission": ["aoss:CreateCollectionItems",
                         "aoss:UpdateCollectionItems",
                         "aoss:DescribeCollectionItems"],
         "ResourceType": "collection"},
        {"Resource": ["index/prod-support-kb/*"],
         "Permission": ["aoss:CreateIndex", "aoss:UpdateIndex",
                         "aoss:DescribeIndex", "aoss:ReadDocument",
                         "aoss:WriteDocument"],
         "ResourceType": "index"}
    ],
    "Principal": [ROLE_ARN, f"arn:aws:iam::{account_id}:root"]
}]
aoss.create_access_policy(
    name='prod-support-kb-access',
    type='data',
    policy=json.dumps(data_policy)
)

# ── Step B: Create Bedrock Knowledge Base ────────────────────────────────
kb = bedrock_agent.create_knowledge_base(
    name='prod-support-kb',
    description='Production support docs and incident history',
    roleArn=ROLE_ARN,
    knowledgeBaseConfiguration={
        'type': 'VECTOR',
        'vectorKnowledgeBaseConfiguration': {
            'embeddingModelArn': (
                'arn:aws:bedrock:us-east-1::foundation-model/'
                'amazon.titan-embed-text-v2:0'
            )
        }
    },
    storageConfiguration={
        'type': 'OPENSEARCH_SERVERLESS',
        'opensearchServerlessConfiguration': {
            'collectionArn': collection_arn,
            'vectorIndexName': 'prod-support-index',
            'fieldMapping': {
                'vectorField': 'embedding',
                'textField': 'text',
                'metadataField': 'metadata'
            }
        }
    }
)
kb_id = kb['knowledgeBase']['knowledgeBaseId']
print(f"Knowledge Base ID: {kb_id}")

# ── Step C: Add S3 data source ───────────────────────────────────────────
ds = bedrock_agent.create_data_source(
    knowledgeBaseId=kb_id,
    name='s3-docs-source',
    dataSourceConfiguration={
        'type': 'S3',
        's3Configuration': {
            'bucketArn': f'arn:aws:s3:::{BUCKET_NAME}'
        }
    },
    vectorIngestionConfiguration={
        'chunkingConfiguration': {
            'chunkingStrategy': 'SEMANTIC',  # keeps incident sections intact
            'semanticChunkingConfiguration': {
                'maxTokens': 512,
                'bufferSize': 1,
                'breakpointPercentileThreshold': 95
            }
        }
    }
)
ds_id = ds['dataSource']['dataSourceId']
print(f"Data Source ID: {ds_id}")

# ── Step D: Start first ingestion job ────────────────────────────────────
job = bedrock_agent.start_ingestion_job(
    knowledgeBaseId=kb_id,
    dataSourceId=ds_id
)
job_id = job['ingestionJob']['ingestionJobId']
print(f"Ingestion job started: {job_id}")

# Poll until complete
while True:
    status = bedrock_agent.get_ingestion_job(
        knowledgeBaseId=kb_id,
        dataSourceId=ds_id,
        ingestionJobId=job_id
    )['ingestionJob']['status']
    print(f"  Ingestion status: {status}")
    if status in ('COMPLETE', 'FAILED'):
        break
    time.sleep(20)

# Save IDs for later scripts
import json
with open('bedrock_ids.json', 'w') as f:
    json.dump({'kb_id': kb_id, 'ds_id': ds_id}, f)
print("IDs saved to bedrock_ids.json")

4Create the Bedrock Agent

The Agent is the orchestration layer. It handles conversation context, retrieves from the Knowledge Base, and generates responses grounded in your documentation.

Python · boto3 — create_agent.py 
import boto3
import json
import time

REGION   = 'us-east-1'
ROLE_ARN = 'arn:aws:iam::ACCOUNT_ID:role/BedrockAgentRole'

with open('bedrock_ids.json') as f:
    ids = json.load(f)
KB_ID = ids['kb_id']

bedrock_agent = boto3.client('bedrock-agent', region_name=REGION)

INSTRUCTION = """You are a production support assistant for our engineering team.
You have access to a knowledge base containing application feature documentation,
historical incident resolution records, and runbooks.
When a support engineer describes an incident or asks a question:
1. Search the knowledge base for relevant context.
2. Provide a concise diagnosis or answer grounded in the retrieved documents.
3. If suggesting resolution steps, number them clearly.
4. If you cannot find relevant information, say so explicitly.
Always cite which document or section informed your answer."""

# 1. Create the agent
agent = bedrock_agent.create_agent(
    agentName='prod-support-triage-bot',
    foundationModel='anthropic.claude-3-sonnet-20240229-v1:0',
    instruction=INSTRUCTION,
    agentResourceRoleArn=ROLE_ARN,
    idleSessionTTLInSeconds=1800,
    description='Production support incident triage and resolution assistant'
)
agent_id = agent['agent']['agentId']
print(f"Agent ID: {agent_id}")

# 2. Associate the Knowledge Base with the agent
bedrock_agent.associate_agent_knowledge_base(
    agentId=agent_id,
    agentVersion='DRAFT',
    knowledgeBaseId=KB_ID,
    description='Production support documentation and incident history',
    knowledgeBaseState='ENABLED'
)
print("Knowledge Base associated.")

# 3. Prepare the agent (compiles the DRAFT version)
bedrock_agent.prepare_agent(agentId=agent_id)
time.sleep(10)  # allow preparation to complete

# 4. Create a versioned snapshot
version_resp = bedrock_agent.create_agent_version(agentId=agent_id)
version_num  = version_resp['agentVersion']['agentVersion']
print(f"Agent version: {version_num}")

# 5. Create a 'production' alias pointing to this version
alias = bedrock_agent.create_agent_alias(
    agentId=agent_id,
    agentAliasName='production',
    routingConfiguration=[{'agentVersion': version_num}]
)
alias_id = alias['agentAlias']['agentAliasId']
print(f"Alias ID: {alias_id}")

# Persist all IDs
ids.update({'agent_id': agent_id, 'alias_id': alias_id})
with open('bedrock_ids.json', 'w') as f:
    json.dump(ids, f, indent=2)
print("All IDs saved to bedrock_ids.json")

5Lambda handler — invoke the agent at runtime

This Lambda function is your chatbot's backend. It uses bedrock-agent-runtime to invoke the agent, maintains session state for multi-turn conversations, and streams the response back to the caller.

Python · boto3 — lambda_handler.py 
import boto3
import json
import os
import uuid
import logging

logger = logging.getLogger()
logger.setLevel(logging.INFO)

runtime = boto3.client('bedrock-agent-runtime',
                         region_name=os.environ.get('AWS_REGION', 'us-east-1'))

AGENT_ID    = os.environ['BEDROCK_AGENT_ID']
AGENT_ALIAS = os.environ['BEDROCK_AGENT_ALIAS_ID']

def lambda_handler(event, context):
    body       = json.loads(event.get('body', '{}'))
    user_msg   = body.get('message', '').strip()
    session_id = body.get('session_id') or str(uuid.uuid4())

    if not user_msg:
        return _response(400, {'error': 'message field is required'})

    try:
        resp = runtime.invoke_agent(
            agentId=AGENT_ID,
            agentAliasId=AGENT_ALIAS,
            sessionId=session_id,
            inputText=user_msg,
            enableTrace=False,
            sessionState={
                'promptSessionAttributes': {
                    'source': 'production-support-chatbot'
                }
            }
        )

        # Stream and concatenate response chunks
        full_text = ''
        for chunk in resp['completion']:
            if 'chunk' in chunk:
                full_text += chunk['chunk']['bytes'].decode('utf-8')

        logger.info({'session': session_id, 'chars': len(full_text)})
        return _response(200, {
            'response': full_text,
            'session_id': session_id
        })

    except Exception as e:
        logger.error(str(e))
        return _response(500, {'error': str(e)})

def _response(status: int, body: dict) -> dict:
    return {
        'statusCode': status,
        'headers': {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*'
        },
        'body': json.dumps(body)
    }
Python · boto3 — deploy_lambda.py

import json

import io

REGION        = 'us-east-1'

6Auto-ingest on new document uploads

An EventBridge rule fires whenever a file lands in the S3 bucket. A Lambda function picks it up and starts a new ingestion job, with debounce logic to avoid duplicate runs.

Python · boto3 — setup_auto_ingest.py (run once) 
import boto3
import json

REGION        = 'us-east-1'
ACCOUNT_ID    = boto3.client('sts').get_caller_identity()['Account']
BUCKET_NAME   = 'prod-support-kb-docs'
FUNCTION_NAME = 'bedrock-kb-ingest'
LAMBDA_ROLE   = 'arn:aws:iam::ACCOUNT_ID:role/LambdaBedrockRole'

with open('bedrock_ids.json') as f:
    ids = json.load(f)

events = boto3.client('events', region_name=REGION)
lmb    = boto3.client('lambda', region_name=REGION)

# ── Deploy the ingestion Lambda ───────────────────────────────────────────
INGEST_CODE = """
import boto3, os, logging
logger = logging.getLogger()
logger.setLevel(logging.INFO)
bedrock = boto3.client('bedrock-agent')
KB_ID = os.environ['KNOWLEDGE_BASE_ID']
DS_ID = os.environ['DATA_SOURCE_ID']

def lambda_handler(event, context):
    running = bedrock.list_ingestion_jobs(
        knowledgeBaseId=KB_ID, dataSourceId=DS_ID,
        filters=[{'attribute': 'STATUS', 'operator': 'EQ',
                  'values': ['IN_PROGRESS', 'STARTING']}]
    )['ingestionJobSummaries']
    if running:
        logger.info("Ingestion already running — skipping.")
        return {'status': 'skipped'}
    job = bedrock.start_ingestion_job(knowledgeBaseId=KB_ID, dataSourceId=DS_ID)
    job_id = job['ingestionJob']['ingestionJobId']
    logger.info(f"Started ingestion job: {job_id}")
    return {'status': 'started', 'jobId': job_id}
"""

import zipfile, io
buf = io.BytesIO()
with zipfile.ZipFile(buf, 'w') as zf:
    zf.writestr('ingest.py', INGEST_CODE)
buf.seek(0)

func = lmb.create_function(
    FunctionName=FUNCTION_NAME,
    Runtime='python3.12',
    Role=LAMBDA_ROLE,
    Handler='ingest.lambda_handler',
    Code={'ZipFile': buf.read()},
    Timeout=60,
    Environment={'Variables': {
        'KNOWLEDGE_BASE_ID': ids['kb_id'],
        'DATA_SOURCE_ID':    ids['ds_id']
    }}
)
func_arn = func['FunctionArn']

# ── Create EventBridge rule for S3 Object Created ─────────────────────────
rule = events.put_rule(
    Name='bedrock-kb-s3-trigger',
    EventPattern=json.dumps({
        "source": ["aws.s3"],
        "detail-type": ["Object Created"],
        "detail": {
            "bucket": {"name": [BUCKET_NAME]},
            "object": {"key": [{"prefix": "incident-history/"},
                                    {"prefix": "app-features/"},
                                    {"prefix": "runbooks/"}]}
        }
    }),
    State='ENABLED'
)

events.put_targets(
    Rule='bedrock-kb-s3-trigger',
    Targets=[{'Id': 'ingest-lambda', 'Arn': func_arn}]
)

# Grant EventBridge permission to invoke the Lambda
lmb.add_permission(
    FunctionName=FUNCTION_NAME,
    StatementId='EventBridgeInvoke',
    Action='lambda:InvokeFunction',
    Principal='events.amazonaws.com',
    SourceArn=rule['RuleArn']
)
print("Auto-ingest pipeline active.")

7Scheduled weekly full re-sync

EventBridge cron schedules a full re-ingestion every Sunday at 02:00 UTC to catch any document updates or deletions that event-driven ingestion may have missed.

Python · boto3 — setup_weekly_sync.py 
import boto3

REGION        = 'us-east-1'
FUNCTION_NAME = 'bedrock-kb-ingest'  # same Lambda as above
ACCOUNT_ID    = boto3.client('sts').get_caller_identity()['Account']

events = boto3.client('events', region_name=REGION)
lmb    = boto3.client('lambda', region_name=REGION)

func_arn = (f'arn:aws:lambda:{REGION}:{ACCOUNT_ID}'
            f':function:{FUNCTION_NAME}')

# Create a cron-based rule: every Sunday at 02:00 UTC
rule = events.put_rule(
    Name='bedrock-kb-weekly-sync',
    ScheduleExpression='cron(0 2 ? * SUN *)',
    State='ENABLED',
    Description='Weekly full re-ingestion for Bedrock KB'
)

events.put_targets(
    Rule='bedrock-kb-weekly-sync',
    Targets=[{'Id': 'weekly-ingest', 'Arn': func_arn}]
)

try:
    lmb.add_permission(
        FunctionName=FUNCTION_NAME,
        StatementId='WeeklyScheduleInvoke',
        Action='lambda:InvokeFunction',
        Principal='events.amazonaws.com',
        SourceArn=rule['RuleArn']
    )
except lmb.exceptions.ResourceConflictException:
    print("Permission already exists — skipping.")

print("Weekly sync rule created: every Sunday 02:00 UTC")

8Agent versioning and deployment via Python

Replace the manual console update or shell scripts with a clean Python deployment script. Run this whenever the agent's system prompt, model, or configuration changes — typically triggered from your CI/CD pipeline as a Python step.

Python · boto3 — deploy_agent_update.py 
import boto3
import json
import time

REGION   = 'us-east-1'
ROLE_ARN = 'arn:aws:iam::ACCOUNT_ID:role/BedrockAgentRole'

with open('bedrock_ids.json') as f:
    ids = json.load(f)
AGENT_ID = ids['agent_id']
ALIAS_ID = ids['alias_id']

# Load the instruction from a file — version-controlled in Git
with open('bedrock/instruction.txt') as f:
    NEW_INSTRUCTION = f.read().strip()

bedrock_agent = boto3.client('bedrock-agent', region_name=REGION)

# 1. Update the agent DRAFT with new instruction / model
bedrock_agent.update_agent(
    agentId=AGENT_ID,
    agentName='prod-support-triage-bot',
    foundationModel='anthropic.claude-3-sonnet-20240229-v1:0',
    instruction=NEW_INSTRUCTION,
    agentResourceRoleArn=ROLE_ARN,
    idleSessionTTLInSeconds=1800
)
print("Agent DRAFT updated.")

# 2. Prepare (compile) the DRAFT
bedrock_agent.prepare_agent(agentId=AGENT_ID)

# Poll until PREPARED
for _ in range(20):
    status = bedrock_agent.get_agent(agentId=AGENT_ID)['agent']['agentStatus']
    print(f"  Agent status: {status}")
    if status == 'PREPARED':
        break
    time.sleep(8)

# 3. Create a new immutable version
version_resp = bedrock_agent.create_agent_version(agentId=AGENT_ID)
new_version  = version_resp['agentVersion']['agentVersion']
print(f"New version created: {new_version}")

# 4. Promote the 'production' alias to the new version
bedrock_agent.update_agent_alias(
    agentId=AGENT_ID,
    agentAliasId=ALIAS_ID,
    agentAliasName='production',
    routingConfiguration=[{'agentVersion': new_version}]
)
print(f"Production alias → version {new_version}")

# 5. Optional: rollback helper
def rollback(target_version: str):
    """Roll the production alias back to a previous version."""
    bedrock_agent.update_agent_alias(
        agentId=AGENT_ID,
        agentAliasId=ALIAS_ID,
        agentAliasName='production',
        routingConfiguration=[{'agentVersion': target_version}]
    )
    print(f"Rolled back to version {target_version}")

# Example: rollback('2')  ← pass previous version number

Python · boto3 — upload_incident.py (post-incident workflow) 
import boto3
from datetime import date

BUCKET_NAME = 'prod-support-kb-docs'
s3 = boto3.client('s3')

def upload_incident_report(filepath: str, service: str, severity: str):
    """
    Upload a resolved incident report to S3.
    EventBridge will fire automatically, triggering re-ingestion.
    """
    today   = date.today().isoformat()
    s3_key  = f'incident-history/{today}-{service}-{severity}.md'

    s3.upload_file(
        filepath,
        BUCKET_NAME,
        s3_key,
        ExtraArgs={
            'ContentType': 'text/markdown',
            'Metadata': {
                'service': service,
                'severity': severity,
                'date': today
            }
        }
    )
    print(f"Uploaded: s3://{BUCKET_NAME}/{s3_key}")
    print("EventBridge will trigger re-ingestion automatically.")

# Usage:
